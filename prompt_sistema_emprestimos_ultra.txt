FUNÇÃO CENTRAL

Criar função:

calculateLoanStatus()

Exemplo:

function calculateLoanStatus(loan) {

const today = new Date() const start = new Date(loan.startDate)

const days = Math.floor((today - start) / (1000 * 60 * 60 * 24))

const interest = loan.amount * 0.01 * days

let fine = 0

if (days > 30) {

const lateDays = Math.min(days - 30, 7)

fine = lateDays * 50

}

const total = loan.amount + interest + fine

return {

days, interest, fine, total

}

}
